# 큐잉모델 시뮬레이션


## 은행 서비스 시뮬레이션 프로그램

<br>

* **소스코드 수정내용**

 **1**. `main()`함수에서의 `service_time`과`servie_customer`을 배열로 바꾸어서 은행창구의 수(`NUMBERS_OF_COUNTER`)만큼 사용 할 수있게 수정하였습니다.

 **2**. 가독성을 위해 출력내용을 조금 수정하였습니다.

```c
#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#define MAX_SIZE 20
#define NUMBERS_OF_COUNTER 2 //창구의 수
typedef struct
{
    int arrival_time;
    int service_time;
    int id;

}element;

typedef struct
{
    int front;
    int rear;
    element data[MAX_SIZE];
}QUEUE;


void error(char* message)
{
    printf(message);
    exit(1);
}

void init_queue(QUEUE* q)
{
    q->front = 0;
    q->rear = 0;
}

int is_empty(QUEUE* q)
{
    return (q->front == q->rear);
}

int is_full(QUEUE* q)
{
    return (q->front == ((q->rear) + 1) % MAX_SIZE);
}


void enqueue(QUEUE* q, element item)
{
    if (is_full(q))
    {
        error("queue is full\n");
        exit(1);
    }
    else
    {
        q->rear = (q->rear + 1) % MAX_SIZE;
        q->data[q->rear] = item;
    }


}

element dequeue(QUEUE* q)
{
    if (is_empty(q))
        error("queue is empty\n");
    else
    {
        q->front = (q->front + 1) % MAX_SIZE;
        return (q->data[q->front]);
    }


}
int main()
{
    int minutes = 60;
    int total_wait = 0;
    int total_customers = 0;
    int service_time[MAX_SIZE] = { 0, };
    int service_customer[MAX_SIZE] = { 0, };
    QUEUE queue;
    init_queue(&queue);

    srand(time(NULL));

    for (int clock = 0; clock < minutes; clock++)
    {
        printf("current time: %d\n", clock);
        if ((rand() % 10) < 3)
        {
            element customer;
            customer.id = total_customers++;
            customer.arrival_time = clock;
            customer.service_time = (rand() % 3) + 1;
            enqueue(&queue, customer);
            printf("(ENTER)      customer %d entered at %2d.       service time required : %d\n",
             customer.id, customer.arrival_time, customer.service_time);
        }
        //수정부분
        for (int i = 1; i < NUMBERS_OF_COUNTER + 1; i++) //창구 수만큼 반복
        {
            if (service_time[i] > 0)// 첫번째 창구부터 비어있는지 확인
            {
                printf("(counter %d)  customer %d is doing business.\n", i, service_customer[i]);
                service_time[i]--;
            }
            else //창구가 비었다면
            {
                if (!is_empty(&queue)) // 대기손님이 있다면 빈 창구에 손님 할당
                {
                    element customer;
                    customer = dequeue(&queue);
                    service_customer[i] = customer.id;
                    service_time[i] = customer.service_time;
                    printf("(counter %d)  customer %d starts his business. he waited %d minutes.\n", 
                    i, customer.id, clock - customer.arrival_time);
                    total_wait = total_wait + clock - customer.arrival_time;
                }
            }
        }
        printf("\n");
    }
    printf("total waiting time : %d \n", total_wait);
    return 0;

}


```
